  <script>
   (() => {
      const alertBox = document.getElementById('authAlert');
      // Vista admin: solo token
      const loginForm = document.getElementById('loginForm');
      const adminTokenInput = document.getElementById('adminToken');

      const showAlert = (variant, message) => {
        if (!alertBox) {
          return;
        }
        alertBox.textContent = message;
        alertBox.className = `alert alert-${variant}`;
      };

      const clearAlert = () => {
        if (!alertBox) {
          return;
        }
        alertBox.textContent = '';
        alertBox.className = 'alert d-none';
      };

      const setLoading = (form, isLoading) => {
        if (!form) {
          return;
        }
        const submitBtn = form.querySelector('button[type="submit"]');
        if (!submitBtn) {
          return;
        }
        if (isLoading) {
          submitBtn.dataset.originalText = submitBtn.textContent;
          submitBtn.textContent = 'Procesando...';
          submitBtn.disabled = true;
        } else {
          submitBtn.disabled = false;
          if (submitBtn.dataset.originalText) {
            submitBtn.textContent = submitBtn.dataset.originalText;
          }
        }
      };

      const parseResponse = async (response) => {
        try {
          return await response.json();
        } catch (error) {
          return {};
        }
      };

      // Registro deshabilitado: no se agrega manejador ni UI

      if (loginForm) {
        loginForm.addEventListener('submit', async (event) => {
          event.preventDefault();
          event.stopPropagation();

          loginForm.classList.add('was-validated');
          if (!loginForm.checkValidity()) {
            return;
          }

          clearAlert();
          setLoading(loginForm, true);

          const adminToken = (adminTokenInput && adminTokenInput.value.trim()) || '';
          if (!adminToken) {
            // Token vacío: mostrar error y re-habilitar botón
            setLoading(loginForm, false);
            showAlert('danger', 'Ingresa el token de administrador.');
            return;
          }

          try {
            const vres = await fetch(`/admin/validate?token=${encodeURIComponent(adminToken)}`);
            const vjson = await parseResponse(vres);
            if (!vres.ok || (vjson && vjson.ok === false)) {
              // Token inválido: mostrar error y re-habilitar botón
              setLoading(loginForm, false);
              showAlert('danger', (vjson && (vjson.error || vjson.message)) || 'Token de administrador inválido.');
              return;
            }
            // OK: redirigir a la vista admin
            window.location.href = `/admin/plans?token=${encodeURIComponent(adminToken)}`;
          } catch (error) {
            // Error de red o servidor: mostrar alerta y re-habilitar
            setLoading(loginForm, false);
            showAlert('danger', 'No se pudo validar el token de administrador.');
          }
        });
      }

      // Vista admin sin toggles
    })();
  </script>
